92 auto lastItr = m_vcmap.lower_bound(static_cast<float>(v));
93 auto nextItr = std::next(lastItr);
94 if (nextItr == m_vcmap.end()) {
95 return lastItr->second;

98 Color from = lastItr->second;
99 Color to = nextItr->second;
100 float interval = nextItr->first - lastItr->first;
101 float percentage = (v - lastItr->first) / interval;
102 return percentage * to + (1.f - percentage) * from;
