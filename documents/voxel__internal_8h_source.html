32 inline StructureMesh* findRegion(entt::registry& reg, entt::entity region, double precision) {

43 inline FloatVDB::Ptr buildMask(const StructureMesh* region, const StructurePoints* input,
44 const openvdb::Vec3d& precision, uint32_t expand)
45 {
46 openvdb::util::NullInterrupter nullInterrupter;
47 openvdb::math::Transform::Ptr xform =
48 openvdb::math::Transform::createLinearTransform();
49 xform->preScale(precision);
50
51 if (!region) {
52 using namespace openvdb;
53
54 auto pmin = Vec3d(std::numeric_limits<double>::max());
55 auto pmax = Vec3d(-std::numeric_limits<double>::max());
56
57 for (const auto& pnt : *input->positions) {
58 pmin.x() = std::min(pnt.x(), pmin.x());
59 pmin.y() = std::min(pnt.y(), pmin.y());
60 pmin.z() = std::min(pnt.z(), pmin.z());
61
62 pmax.x() = std::max(pnt.x(), pmax.x());
63 pmax.y() = std::max(pnt.y(), pmax.y());
64 pmax.z() = std::max(pnt.z(), pmax.z());
65 }
66
67 Vec3i cmin = xform->worldToIndex(pmin);
68 Vec3i cmax = xform->worldToIndex(pmax);
69
70 auto bbox = openvdb::CoordBBox(
71 openvdb::Coord(cmin), openvdb::Coord(cmax));
72 bbox.expand(expand);
73
74 openvdb::FloatGrid::Ptr ret = openvdb::FloatGrid::create(0.f);
75 ret->denseFill(bbox, 1.f);
76 ret->setTransform(xform);
77 return ret;
78 }
79 else {
80
81 VectorVec3<float> floatVs;
82 floatVs.resize(region->vertices->size());
83 for (size_t i = 0, I = floatVs.size(); i < I; i++)
84 {
85 floatVs[i] = (*region->vertices)[i];
86 }
87
88 if (region->quadrangles) {
89 return openvdb::tools::doMeshConversion<openvdb::FloatGrid>(nullInterrupter,
90 *xform, floatVs, *region->triangles, *region->quadrangles,
91 (float)expand, std::numeric_limits<float>::max());
92 }
93 else {
94 return openvdb::tools::doMeshConversion<openvdb::FloatGrid>(nullInterrupter,
95 *xform, floatVs, *region->triangles, {},
96 (float)expand, std::numeric_limits<float>::max());
97 }
98 }
99
100 return nullptr;
101 }
