220 template <class TreeType>
221 inline void
222 MinMaxVoxel<TreeType>::operator()(const tbb::blocked_range<size_t>& range)
223 {
224 typename TreeType::LeafNodeType::ValueOnCIter iter;
225
226 for (size_t n = range.begin(); n < range.end(); ++n) {
227 iter = mLeafArray.leaf(n).cbeginValueOn();
228 for (; iter; ++iter) {
229 const ValueType value = iter.getValue();
230 mMin = std::min(mMin, value);
231 mMax = std::max(mMax, value);
232 }
233 }
